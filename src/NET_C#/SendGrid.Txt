    public class EmailService : IEmailService
    {
        private IWebHostEnvironment _env;
        private AppKeys _appKeys;

        public EmailService(IWebHostEnvironment env, IOptions<AppKeys> appKeys)
        {
            _env = env;
            _appKeys = appKeys.Value;
        }

        public async void Add(EmailsAddRequest model)
        {
            var path = "/EmailTemplate/Test.html";
            var filePath = _env.WebRootPath + path;
            var htmlContent = System.IO.File.ReadAllText(filePath).Replace("{{&&message}}", "Welcome!");
            SendGridMessage message = new SendGridMessage()
            {
                From = new EmailAddress("devteam@gmail.com", "Dev Team"),
                Subject = model.Subject,
                PlainTextContent = model.PlainTextContent,
                HtmlContent = htmlContent
            };
            message.AddTo(model.To);
            await Send(message);
        }

        private async Task Send(SendGridMessage message)
        {
            var client = new SendGridClient(_appKeys.SendGridAppKey);
            await client.SendEmailAsync(message);
           
        }
